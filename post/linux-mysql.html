MySQL的安装

MySQL数据库服务是一个完全托管的数据库服务,可使用世界上最受欢迎的开源数据库来部署云原生应用程序。

# 1. 设置MySQL软件源

echo "deb http://repo.mysql.com/apt/debian/ buster mysql-5.7" | tee /etc/apt/sources.list.d/mysql.list
1

# 2. 创建数据库存储目录

mkdir -p /data/mysql/prod
1

# 3.导入密钥更新软件源并安装数据库

# 4. 修改配置文件

修改/etc/mysql/my.cnf配置文件，可根据实际情况修改对应的配置，例如存储位置datadir之类的参数，如下快捷命令可直接执行

echo '!includedir /etc/mysql/conf.d/

[mysqld]
# 基础信息配置
port=3306
character_set_server=utf8mb4
datadir=/data/mysql/prod
socket=/var/run/mysqld/mysqld.sock
symbolic-links=0
log-error=/var/log/mysql/error.log
pid-file=/var/run/mysqld/mysqld.pid
sql_mode=STRICT_TRANS_TABLES,NO_ZERO_IN_DATE,NO_ZERO_DATE,ERROR_FOR_DIVISION_BY_ZERO,NO_AUTO_CREATE_USER,NO_ENGINE_SUBSTITUTION
# 区分大小写配置 0：表名敏感字段不敏感 1：表名不敏感字段敏感 2：都不敏感
lower_case_table_names=0

# 连接配置
max_connections=3000
max_user_connections=3000
max_connect_errors=100000
max_allowed_packet=1G
collation_server=utf8mb4_general_ci
table_open_cache=8000

# 日志配置
expire_logs_days=30
binlog_format=mixed
slow_query_log=ON
slow_query_log_file=/data/mysql/prod/slow.log
long_query_time=3
log_queries_not_using_indexes=on

[mysql]
default_character_set=utf8mb4
[client]
default_character_set=utf8mb4' > /etc/mysql/my.cnf

# 5. 启动MySQL并设置开机自启动

systemctl restart mysql && systemctl enable mysql
1

# 6. 登录MySQL

mysql -u root -p
1

# 7. 创建数据库和用户

-- .表示授权所有数据库和所有表，'用户名'表示要授权的用户名，'%'表示任意主机IP都可以访问，'密码'表示用户登录密码。
GRANT ALL PRIVILEGES ON *.* TO '用户名'@'%' IDENTIFIED BY '密码';

-- 如果要授权特定主机IP访问，可以将'%'替换为具体的IP地址
GRANT ALL PRIVILEGES ON *.* TO '用户名'@'192.168.1.100' IDENTIFIED BY '密码';

-- 如果只想让用户访问特定数据库或表，可以将*.*替换为具体的数据库或表名
GRANT ALL PRIVILEGES ON `mydb`.* TO '用户名'@'%' IDENTIFIED BY '密码';

-- 如果想授权用户对数据库执行某些操作，如只能查询、插入或删除数据
GRANT SELECT, INSERT, DELETE ON `mydb`.* TO '用户名'@'%' IDENTIFIED BY '密码';

-- 刷新权限
FLUSH PRIVILEGES;
